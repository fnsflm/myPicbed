2.2 实验二

2.2.1实验内容

调度问题, 给定n个任务和它们完成所消耗的时间, 求合理的调度方案, 求出最短平均消耗时间. 这里有4个任务, 完成任务的消耗时间分别为15, 8 , 3, 10.

输入: 第一行整数n个任务, 之后n行为每个任务所消耗的时间

2.2.2 实验过程

使用贪心算法, 任务时间短的先执行, 让尽可能多的人等待尽可能等待更少的时间, 对时间进行排序, 即可计算最终的平均消耗时间.

xxxxxxxxxx
#include<iostream>
#include<algorithm>
using namespace std;
int main(int argc, char const *argv[])
{
    int n;
    cin >> n;
    int *job_time = new int[n];
    for (int i = 0; i < n; i ++){
        cin >> job_time[i];
    }
    sort(job_time, job_time + n);
    int cost = 0;
    for(int i = 0; i < n; i ++){
        cost += job_time[i] * (n - i);
    }
    cout << (cost + 0.0) / n << endl;
    return 0;
}

2.3.2 实验过程

bellman-ford算法:

xxxxxxxxxx
#include <cstdio>
#include <algorithm>
using namespace std;
#define INF 0x7FFFFFFF
int G[500][500], Dist[500], Path[500], nPath[500];
bool check( int N )
{
    bool flag = false;
    for( int i = 1; i <= N; ++i )
        for( int j = 1; j <= N; ++j )
            if( G[i][j] != INF && Dist[i] != INF && Dist[j] >= Dist[i] + G[i][j] )
            {
                if( Dist[j] > Dist[i] + G[i][j] )
                {
                    flag = true;
                    Dist[j] = Dist[i] + G[i][j];
                    Path[j] = i;
                    nPath[j] = nPath[i];
                }
                else if( i != Path[j] )
                    nPath[j] += nPath[i];
            }
    return flag;
}
bool BF( int N )
{
    for( int i = 1; i <= N - 1; ++i )
        if( !check(N) )
            return true;
    return !check(N);
}
int main()
{
    int N, M, V1, a, b, w;
    scanf("%d %d %d", &N, &M, &V1);
    fill( G[0], G[0] + 500 * 500, INF );
    fill( Dist, Dist + N  + 1, INF );
    Dist[V1] = 0;
    for( int i = 1; i <= M; ++i )
    {
        scanf("%d %d %d", &a, &b, &w);
        G[a][b] = w;
    }
    if( BF( N ) )
    {
        for( int i = 1; i <= N; ++i )
            printf("%d ", Dist[i]);
    }
    else    printf("FALSE");
}

2.3.3 实验结果

输出为指定起点到各个点的最短路径

2.3.4 实验小结

bellman-ford适合处理这种带负权值的单源点的最短路径问题, 以松弛操作为基础，即估计的最短路径值渐渐地被更加准确的值替代，直至得到最优解。计算时每个边之间的估计距离值都比真实值大，并且被新找到路径的最小长度替代。在重复地计算中，已计算得到正确的距离的边的数量不断增加，直到所有边都计算得到了正确的路径。这样的策略使得贝尔曼-福特算法比迪科斯彻算法适用于更多种类的输入。

#include <cstdio>
#include <algorithm>
using namespace std;
int G[500][500];
#define INF 0x7FFFFFFF
int main(int argc, char const *argv[])
{
    int N, M, V1, a, b, w;
    scanf("%d %d %d", &N, &M, &V1);
    fill(G[0], G[0] + 500 * 500, INF);
    for (int i = 1; i <= N; i++)
    {
        G[i][i] = 0;
    }
    for (int i = 1; i <= M; ++i)
    {
        scanf("%d %d %d", &a, &b, &w);
        G[a][b] = w;
    }
    for (int k = 1; k <= N; k++)
    {
        for (int i = 1; i <= N; i++)
        {
            for (int j = 1; j <= N; j++)
            {
                if (G[i][k] != INF && G[k][j] != INF)
                    if (G[i][k] + G[k][j] < G[i][j])
                        G[i][j] = G[i][k] + G[k][j];
            }
        }
    }
    for (int i = 1; i <= N; i++)
    {
        printf("%d ", G[V1][i]);
    }
    return 0;
}
